📝 การเขียน Test Case API ด้วย Postman

🎯 สิ่งที่ต้องตรวจสอบใน API Test

หัวข้อ รายละเอียด ตัวอย่าง
Status Code ตรวจสอบ HTTP status ที่ตอบกลับ 200, 201, 400, 404, 500
Response Time เวลาตอบสนองต้องรวดเร็ว < 500ms, < 2s
Response Body ตรวจสอบ JSON structure และค่า มี field ครบ, ค่าถูกต้อง
Headers ตรวจสอบ Content-Type และ headers อื่นๆ application/json
Schema Validation ตรวจสอบว่าข้อมูลตรงกับ schema ที่กำหนด type, required fields
Error Handling ตรวจสอบกรณี error invalid input, not found

🛍️ ตัวอย่าง: E-commerce API (Test Shop)

💡 API Endpoints ที่ต้องทดสอบ

  • GET /api/products - ดึงรายการสินค้า
  • GET /api/products/{id} - ดึงสินค้าตาม ID
  • POST /api/cart - เพิ่มสินค้าในตะกร้า
  • POST /api/orders - สร้างคำสั่งซื้อ
  • GET /api/orders/{id} - ดึงรายละเอียดคำสั่งซื้อ

📋 Test Case Template สำหรับ Postman

Test Case 1: GET Products List

// Test Case: Get All Products
// Method: GET
// URL: {{base_url}}/api/products

// 1. ตรวจสอบ Status Code
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

// 2. ตรวจสอบ Response Time
pm.test("Response time is less than 500ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(500);
});

// 3. ตรวจสอบ Content-Type
pm.test("Content-Type is application/json", function () {
    pm.response.to.have.header("Content-Type");
    pm.expect(pm.response.headers.get("Content-Type")).to.include("application/json");
});

// 4. ตรวจสอบ Response Body เป็น Array
pm.test("Response is an array", function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.be.an('array');
});

// 5. ตรวจสอบว่ามีสินค้าอย่างน้อย 1 รายการ
pm.test("Products array is not empty", function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData.length).to.be.above(0);
});

// 6. ตรวจสอบ Schema ของสินค้าแต่ละรายการ
pm.test("Each product has required fields", function () {
    const jsonData = pm.response.json();
    jsonData.forEach(function(product) {
        pm.expect(product).to.have.property('id');
        pm.expect(product).to.have.property('name');
        pm.expect(product).to.have.property('price');
        pm.expect(product).to.have.property('stock');
        
        // ตรวจสอบ type
        pm.expect(product.id).to.be.a('number');
        pm.expect(product.name).to.be.a('string');
        pm.expect(product.price).to.be.a('number');
        pm.expect(product.stock).to.be.a('number');
        
        // ตรวจสอบค่า
        pm.expect(product.price).to.be.above(0);
        pm.expect(product.stock).to.be.at.least(0);
    });
});

// 7. เก็บ product ID ตัวแรกไว้ใช้ test ต่อไป
const firstProduct = pm.response.json()[0];
pm.collectionVariables.set("product_id", firstProduct.id);
pm.collectionVariables.set("product_price", firstProduct.price);

Test Case 2: GET Product by ID

// Test Case: Get Product by ID
// Method: GET
// URL: {{base_url}}/api/products/{{product_id}}

// 1. ตรวจสอบ Status Code
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

// 2. ตรวจสอบ Response Body
pm.test("Response is an object", function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.be.an('object');
});

// 3. ตรวจสอบข้อมูลสินค้าถูกต้อง
pm.test("Product has correct ID", function () {
    const jsonData = pm.response.json();
    const expectedId = pm.collectionVariables.get("product_id");
    pm.expect(jsonData.id).to.eql(parseInt(expectedId));
});

// 4. ตรวจสอบ Required Fields
pm.test("Product has all required fields", function () {
    const product = pm.response.json();
    pm.expect(product).to.have.all.keys('id', 'name', 'description', 'price', 'stock', 'category');
});

Test Case 3: POST Add to Cart

// Test Case: Add Product to Cart
// Method: POST
// URL: {{base_url}}/api/cart
// Body (JSON):
// {
//     "product_id": {{product_id}},
//     "quantity": 2
// }

// 1. ตรวจสอบ Status Code
pm.test("Status code is 201 (Created)", function () {
    pm.response.to.have.status(201);
});

// 2. ตรวจสอบ Response Body
pm.test("Response contains cart item", function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('cart_id');
    pm.expect(jsonData).to.have.property('product_id');
    pm.expect(jsonData).to.have.property('quantity');
    pm.expect(jsonData).to.have.property('subtotal');
});

// 3. ตรวจสอบการคำนวณราคา
pm.test("Subtotal is calculated correctly", function () {
    const jsonData = pm.response.json();
    const productPrice = pm.collectionVariables.get("product_price");
    const expectedSubtotal = productPrice * jsonData.quantity;
    pm.expect(jsonData.subtotal).to.eql(expectedSubtotal);
});

// 4. เก็บ Cart ID
pm.collectionVariables.set("cart_id", pm.response.json().cart_id);

Test Case 6: Error Handling - Validation Error

// Test Case: Create Order with Invalid Data
// Method: POST
// URL: {{base_url}}/api/orders
// Body (JSON) - Missing required fields:
// {
//     "cart_id": "",
//     "shipping_address": {}
// }

// 1. ตรวจสอบ Status Code 400
pm.test("Status code is 400 (Bad Request)", function () {
    pm.response.to.have.status(400);
});

// 2. ตรวจสอบ Validation Error
pm.test("Response contains validation errors", function () {
    const jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('errors');
    pm.expect(jsonData.errors).to.be.an('array');
});

// 3. ตรวจสอบ Error Messages
pm.test("Error messages indicate missing fields", function () {
    const errors = pm.response.json().errors;
    const errorMessages = errors.map(e => e.message.toLowerCase());
    pm.expect(errorMessages).to.include.members(['cart_id is required', 'shipping address is required']);
});

🔧 Pre-request Script (ตัวอย่าง)

// Pre-request Script - ทำก่อนส่ง request
// ใช้สำหรับเตรียมข้อมูล เช่น สร้าง timestamp, random data

// 1. สร้าง Unique Email
const timestamp = new Date().getTime();
const uniqueEmail = `test${timestamp}@example.com`;
pm.collectionVariables.set("unique_email", uniqueEmail);

// 2. สร้าง Random Quantity (1-5)
const randomQty = Math.floor(Math.random() * 5) + 1;
pm.collectionVariables.set("random_quantity", randomQty);

// 3. ตั้งค่า Header ถ้าต้องการ
pm.request.headers.add({
    key: 'X-Request-ID',
    value: pm.variables.replaceIn('{{$guid}}')
});

// 4. Console Log ดูค่า
console.log("Generated Email:", uniqueEmail);
console.log("Random Quantity:", randomQty);

📊 Postman Collection Structure ที่แนะนำ

Test Shop API
├── Setup
│   └── POST Login (เก็บ auth token)
├── Products
│   ├── GET /products (List all)
│   ├── GET /products/{id} (Get one)
│   └── GET /products/{id} (404 test)
├── Cart
│   ├── POST /cart (Add item)

│   └── DELETE /cart/{id} (Remove item)
├── Orders
│   ├── POST /orders (Create - success)
│   ├── POST /orders (Create - validation error)

│   └── PATCH /orders/{id}/cancel (Cancel order)
└── Cleanup
    └── DELETE /cart/clear

✅ Checklist ก่อน Run Collection

  • ตั้งค่า Environment variables (base_url, auth_token)
  • ตรวจสอบว่า API server ทำงานปกติ
  • มีข้อมูล test ใน database (products, users)
  • ตรวจสอบ authentication token ยังไม่หมดอายุ
  • ตั้งค่า Iteration ถ้าต้องการรันหลายรอบ
  • เปิด Console เพื่อดู log ถ้ามี error
